414 static void IntToUnicode(uint32_t value, uint8_t * pbuf, uint8_t len)
415 {
416  uint8_t idx = 0;
417 
418  for (idx = 0; idx < len; idx++)
419  {
420  if (((value >> 28)) < 0xA)
421  {
422  pbuf[2 * idx] = (value >> 28) + '0';
423  }
424  else
425  {
426  pbuf[2 * idx] = (value >> 28) + 'A' - 10;
427  }
428 
429  value = value << 4;
430 
431  pbuf[2 * idx + 1] = 0;
432  }
